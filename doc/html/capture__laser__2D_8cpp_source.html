459 Eigen::VectorXs CaptureLaser2D::computeSquaredMahalanobisDistances(const FeatureBase* _feature_ptr, const Eigen::Vector4s& _expected_feature, const Eigen::Matrix3s& _expected_feature_cov, const Eigen::MatrixXs& _mu)
460 {
461  assert(_mu.rows() == 3 && "mahalanobis distance with bad number of mu components");

463  const Eigen::Vector2s& p_feature = _feature_ptr->getMeasurement().head(2);
464  const WolfScalar& o_feature = _feature_ptr->getMeasurement()(2);

466  // ------------------------ d
467  Eigen::Vector3s d;
468  d.head(2) = p_feature - _expected_feature.head(2);

476 
477  Eigen::VectorXs squared_mahalanobis_distances(_mu.cols());
478 
479  for (unsigned int i = 0; i < _mu.cols(); i++)
480  squared_mahalanobis_distances(i) = (d - _mu.col(i)).transpose() * iSigma_d * (d - _mu.col(i));

570 void CaptureLaser2D::createCornerLandmark(FeatureCorner2D* _corner_ptr, const Eigen::Vector3s& _feature_global_pose)

574  getTop()->addState(new_landmark_state_position, _feature_global_pose.head(2));
575  StateOrientation* new_landmark_state_orientation = new StateTheta(getTop()->getNewStatePtr());
576  getTop()->addState(new_landmark_state_orientation, _feature_global_pose.tail(1));
577 
578  // Initialize landmark covariance
579  Eigen::MatrixXs Sigma_robot = Eigen::MatrixXs::Zero(3,3);

595  new_landmark, //landmark pointer
596  getFramePtr()->getPPtr(), //_robotPPtr,
597  getFramePtr()->getOPtr(), //_robotOPtr,
598  new_landmark->getPPtr(), //_landmarkPPtr,
